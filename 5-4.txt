#include <stdio.h>
#include <iostream>
#include <string>
#include <vector>

#include <ctime>

using namespace std;

char& find(char a[], char c, bool& success){
    for(int i=0; i<sizeof(a); i++){
        if(a[i] == c){
            success = true;
            return a[i];
        }
        a++;
    }
    success = false;
    return c;
};

int main() {
    char s[] = "Mike";
    bool b = false;
    char& loc = find(s, 'M', b);
    if (b == false) {
        cout << "M을 발견할 수 없다" << endl;
        return 0;
    }
    loc = 'm';
    cout << s << endl;
}
