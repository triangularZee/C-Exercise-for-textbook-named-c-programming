#include <stdio.h> // 전처리가 중요. 이 문제에서는 toLower()
#include <iostream>
#include <string>
#include <vector>
#include <sstream>

using namespace std;

class Histogram{
    string str;
    int Alphabet[26];
    
public:
    Histogram(string s) { str = s; };
    void put(string s) { str.append(s); };
    void putc(char c) { str.push_back(c); };
    void print();
    void AlphaCount();
    int CountALL();
};

void Histogram::print() {
    cout << str << endl << endl;
    for(int i=0; i<str.size(); i++){ //LowerTo()
        if('A'<=str[i] && str[i]<='Z'){
            str[i] = char(str[i]+32);
        }
    }
    cout << "총 알파벳 수 : " << CountALL() << endl << endl;
    AlphaCount();
}

void Histogram::AlphaCount() {
    char alpha = 'a';
    int count = 0;
    while(alpha <= 'z'){
        cout << alpha << '\t';
        for(int i=0; i<str.size(); i++){
            if(str[i] == alpha){
                count++;
            }
        }
        cout << '(' << count << ')' << '\t' << ':' << ' ';
        for(int j=0; j<count; j++){
            cout << '*';
        }
        alpha++;
        count = 0;
        cout << endl;
    }
}

int Histogram::CountALL() {
    int num = 0;
    for(int i=0; i<str.size(); i++){
        if(isalpha(str[i])){
            num++;
        }
    }
    return num;
}
